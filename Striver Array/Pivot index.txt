class Solution {
    public int pivotIndex(int[] nums) {
        int total = 0;
        for (int val : nums) total += val;

        int left = 0;
        for (int i = 0; i < nums.length; i++) {
            total -= nums[i];      // remove current element from right side

            if (left == total) {   // check equality
                return i;
            }

            left += nums[i];       // add current element to left side
        }
        return -1;
    }
}
